155      //Retrieves the forecast from the page url passed to it
156      class getForecastAsync extends AsyncTask<String,String,String> {
157          @Override
158          protected void onPreExecute() {
159              super.onPreExecute();
160              Toast.makeText(MainActivity.this, "Retrieving forecast",Toast.LENGTH_LONG).show();
161              Log.d("ForecastObj","Retrieving forecast");
162          }
163  
164          @Override
165          protected void onPostExecute(String result) {
166              super.onPostExecute(result);
167              Toast.makeText(MainActivity.this, "The connection result: " + result, Toast.LENGTH_LONG).show();
168              Log.d("ForecastObj","The connection result: " + result);
169          }
170  
171          @Override
172          protected String doInBackground(String... strings) {
173              String result;
174              final String TAG = "doInBackground";
175  
176              try {
177                  String url =strings[0];
178                  //parse the page html to the document
179                  Document pageHtml = Jsoup.connect(url).get();
180                  //Check document is not empty
181                  if (pageHtml != null) {
182                      Elements listElement = pageHtml.select("ul.pollen_graph");
183                      final String forecasts = ""+ listElement.select("li").text();
184                      result = "successful\n Values:"+forecasts;//**
185                      //split the pollen forecast values into array (potential values: Low, Moderate, High, Very High, Extreme)
186                      String delim = "\\W+";
187                      String[] days = forecasts.split(delim);
188                      ArrayList<String> daysList = new ArrayList<>();
189                      Collections.addAll(daysList, days);
190                      forecastObj.setDaysList(daysList);//update daysList in forecastObj
191                  }
192                  else {
193                      result = "could not retrieve page from site\nUrl: "+url;
194                  }
195              }//end of try
196              catch (Exception ex) {
197                  ex.printStackTrace();
198  //                Log.e(TAG,"An Exception was thrown\n" +ex);
199                  Log.e("ForecastObj","An Exception was thrown\n" +ex);
200                  result = "An Exception was thrown\n"+ex;
201              }//end of catch
202              return result;//returns result to onPostExecute
203          }//end of doInBackground
204      }//end of getForecastAsync
